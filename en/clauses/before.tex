\section{Before Your First Day}
\label{sec:before}

\subsection{Confirm the Basics}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Start time and expected arrival (aim for 10–15 minutes early) & $\square$ Office location or remote login details \\ 
$\square$ Parking, building access, or security procedures & $\square$ Dress code (ask HR if unclear) \\ 
\end{tabular}

\subsection{Review Company Information}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Revisit the company website & $\square$ Review mission, vision, and core values \\ 
$\square$ Skim recent news or announcements & $\square$ Review job description and responsibilities \\ 
\end{tabular}

\subsection{Prepare Your Materials}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Government ID (if required) & $\square$ Banking info for payroll \\ 
$\square$ Notepad or digital note app & $\square$ List of prepared questions \\ 
$\square$ Lunch/snacks (if unsure of options nearby) & \\ 
\end{tabular}

\subsection{Mental Preparation}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Set realistic expectations (you won’t know everything yet) & $\square$ Prepare a short self-introduction \\ 
$\square$ Plan 2–3 questions about your team and role & $\square$ Get a full night’s sleep \\ 
\end{tabular}
