\section{During Your First Day}
\label{sec:during}

\subsection{Arrival and First Impressions}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Arrive early & $\square$ Greet people confidently \\ 
$\square$ Introduce yourself clearly (name + role) & $\square$ Maintain positive body language \\ 
\end{tabular}

\subsection{Orientation and HR Tasks}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Complete paperwork (tax, benefits, policies) & $\square$ Set up payroll and benefits \\ 
$\square$ Review employee handbook & $\square$ Get ID badge, keys, or equipment \\ 
$\square$ Set up email and required accounts & \\ 
\end{tabular}

\subsection{Meet Your Manager}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Ask questions & $\square$ Clarify expectations \\ 
$\square$ Confirm reporting structure & $\square$ Schedule recurring check-ins \\ 
\end{tabular}

\subsection{Meet Your Team}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Learn names and roles & $\square$ Ask what they’re currently working on \\ 
$\square$ Understand how your role supports theirs & $\square$ Take notes on key contacts \\ 
\end{tabular}

\subsection{Understand Tools and Processes}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Learn core software platforms & $\square$ Ask about workflows and approvals \\ 
$\square$ Understand communication channels (Slack, Teams, email) & $\square$ Review ongoing projects \\ 
\end{tabular}

\subsection{Observe Company Culture}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Notice how meetings are run & $\square$ Note work pace and expectations \\ 
$\square$ Pay attention to decision-making patterns & $\square$ Observe communication style (formal/informal) \\ 
\end{tabular}
